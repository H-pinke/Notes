#include <stdlib.h>
#include <stdbool.h>

bool sum() {
    int i;
    int num;
    num = num+6;
    i = num;
    printf("%d\n", i);
    return true;
} 
int num() {
    return 1;
}
int main()
{
    
    sum();
    //printf("Hello, World");
    num();
